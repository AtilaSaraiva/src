\[
\frac{1}{v^2} \ddot{\UU} = 
\rho \DIV{\lp \frac{1}{\rho} \GRAD{\UU} \rp} -f
\]

\sep

\begin{itemize}
   \item \textbf{modeling}: propagate waves forward
   \item \textbf{migration}: propagate waves backward 
\end{itemize}

\end{frame}
